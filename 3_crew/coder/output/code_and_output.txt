# Python Program to Calculate the Series
total = 0
for i in range(10000):
    term = 1 / (2 * i + 1)  # Calculate the term as 1/(2*i + 1)
    if i % 2 == 0:
        total += term  # Add when i is even
    else:
        total -= term  # Subtract when i is odd

result = total * 4  # Multiply the total by 4
print(result)  # Print the result

# Output:
3.1414926535900345
